// Server rate change event script
// Increases rates to 2x every Friday at 5 PM, and returns to 1x every Monday at 5 AM

-	script	RateChanger	-1,{

OnInit:
	// To force the event to run even if you have to shutdown your server for some reason...
	if (gettime(4) != 5 && gettime(4) != 6 && gettime(4) != 0) end;

    // Schedule rate change on Fridays at 5 PM (17:00) and Monday at 5 AM (05:00)
    OnFri1700:
        setBattleFlag "base_exp_rate", 200, true;
        setBattleFlag "job_exp_rate", 200, true;
		setBattleFlag "item_rate_common", 125, true;
		setBattleFlag "item_rate_common_boss", 125, true;
		setBattleFlag "item_rate_common_mvp", 125, true;
		setBattleFlag "item_rate_heal", 125, true;
		setBattleFlag "item_rate_heal_boss", 125, true;
		setBattleFlag "item_rate_heal_mvp", 125, true;
		setBattleFlag "item_rate_use", 125, true;
		setBattleFlag "item_rate_use_boss", 125, true;
		setBattleFlag "item_rate_use_mvp", 125, true;
		setBattleFlag "item_rate_equip", 125, true;
		setBattleFlag "item_rate_equip_boss", 125, true;
		setBattleFlag "item_rate_equip_mvp", 125, true;
		setBattleFlag "item_rate_card", 125, true;
		setBattleFlag "item_rate_card_boss", 125, true;
		setBattleFlag "item_rate_card_mvp", 125, true;
		announce "Floating Rates Enabled!", bc_all;
        end;

    OnMon0500:
        setBattleFlag "base_exp_rate", 100, true;
        setBattleFlag "job_exp_rate", 100, true;
		setBattleFlag "item_rate_common", 100, true;
		setBattleFlag "item_rate_common_boss", 100, true;
		setBattleFlag "item_rate_common_mvp", 100, true;
		setBattleFlag "item_rate_heal", 100, true;
		setBattleFlag "item_rate_heal_boss", 100, true;
		setBattleFlag "item_rate_heal_mvp", 100, true;
		setBattleFlag "item_rate_use", 100, true;
		setBattleFlag "item_rate_use_boss", 100, true;
		setBattleFlag "item_rate_use_mvp", 100, true;
		setBattleFlag "item_rate_equip", 100, true;
		setBattleFlag "item_rate_equip_boss", 100, true;
		setBattleFlag "item_rate_equip_mvp", 100, true;
		setBattleFlag "item_rate_card", 100, true;
		setBattleFlag "item_rate_card_boss", 100, true;
		setBattleFlag "item_rate_card_mvp", 100, true;
        announce "Floating Rates Disabled", bc_all;
        end;

}
